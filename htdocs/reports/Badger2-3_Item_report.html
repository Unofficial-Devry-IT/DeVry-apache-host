"""
Item
   usable / consumable
   object / weapon
   title / name
   description
   size / weight
   constraints on usability
"""


class Item:
    def __init__(self, **kwargs):
        """
        Stores basic information regarding items
            name (str): text that represents item
            weight (float): how heavy the item is
            desc (str): description of item
        :param kwargs:
        """
        self._name = kwargs.pop("name", "DefaultName")
        self._weight = kwargs.pop("weight", 0.0)
        self._desc = kwargs.pop("desc", "")
        self._stackable = kwargs.pop("stackable", False)
        self._item_count = kwargs.pop("stack_count", 1)
        self._stack_size = kwargs.pop("stack_size", 1)
        self._price = kwargs.pop("price", 0.01)

        if len(kwargs) > 0:
            raise Exception("ITEM DOES NOT USE ALL PARAMETERS: ", kwargs)

    def increase_stack(self, amount=1):
        self._item_count += amount

        if self._item_count > self._stack_size:
            self._item_count = self._stack_size

    def decrease_stack(self, amount=1):
        self._item_count -= amount

        if self._item_count < 0:
            self._item_count = 0

    @property
    def exhausted(self):
        return self._item_count == 0

    @property
    def name(self):
        return self._name

    @property
    def weight(self):
        return self._weight

    @property
    def desc(self):
        return self._desc

    @property
    def is_stackable(self):
        return self._stackable

    @property
    def item_count(self):
        return self._item_count

    @property
    def at_max_stack(self):
        return self.item_count == self._stack_size

    @property
    def price(self):
        return self._price
